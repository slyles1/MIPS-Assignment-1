# MIPS Assignment 1
.data
	str: .space 32
	message1: .asciiz "\nEnter your 10 character string"
	newline: .asciiz "\n"
	


.text # for instructions
main:
	
	# prints message1
	li $v0, 4
	la $a0, message1
	syscall
	
	# read a string to operate on
	li $v0, 8
	la $a0, str
	li $a1, 11 # space for 11 characters
	syscall
	
	# print out new line
	li $v0, 4
	la $a0, newline
	syscall


	exit:
	li $v0, 0
	syscall
