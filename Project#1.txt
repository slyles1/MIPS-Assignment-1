.data

	message3: .asciiz " inside the outside label\n"

.text # for instructions

	# prints message1
	li $v0, 4

	blt $t1, 1, exit # Branch less than 10 < 1
	
    lbu $a0, 0($t7) # Loads byte 0 of $t7 (str)

    # Algorithm for checking characters
    li $t0, 0

    # false

    # print out message2

	la $a0, message2

    addi $t7, 1 # next char
	add $t2, $t2, 1 # Adds 1 to $t2 to get the next character
	sub $t1, $t1, 1 # --10

	# Exit call
	li $v0, 0
	syscall

	# False
	# Conversion and summation
	sub $t9, $a0, 48 # 48 - 48 is 0 in decimal
	add $k0, $k0, $t9 # adds character to total sum $k0
	
	# print out colon
	li $v0, 4
	la $a0, colon
	syscall
	
	# print int
	li $v0, 1
	add $a0, $zero, $k0 # put 0 + $k0 in argument register
	syscall
	
	# print out new line
	li $v0, 4
	la $a0, newline
	syscall
	
	j return
	
	checkA:
	li $t0, 84 #ASCII after S
	bge $a0, $t0, checkS # character index of string >= 84 if true run checkA
	
	# False

	syscall
	
	
	# print int
	li $v0, 1
	add $a0, $zero, $k0 # put 0 + $k0 in argument register
	syscall
	
	# print out new line
	li $v0, 4
	la $a0, newline

	j return
