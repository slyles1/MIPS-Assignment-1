# MIPS Assignment 1
.data
	prompt: .asciiz "Enter your Base number: "
	message: .asciiz "\nYour Base is "
	message1: .asciiz "\nThe end of your base is the nth letter "
	alph: .space 32


.text # for instructions
main:
	li $v0, 4 # Printing a string and system call code for printing string
	la $a0, prompt # load address of string to be printed into $a0
	syscall

	li $v0, 5 # get an integer
	syscall

	move $t0, $v0 # store integer
	
	# Display "Your Base is"
	li $v0, 4
	la $a0, message
	syscall

	# Print result in $t0
	li $v0, 1
	move $a0, $t0
	syscall

	# base - 10
	li $t1, 10
	sub $t0, $t0, $t1
	
	li $v0, 4
	la $a0, message1
	syscall
	
	# Print result in $t0 (the ending nth letter of the base)
	li $v0, 1
	move $a0, $t0
	syscall
	


	# Exit call
	li $v0, 0
	syscall
